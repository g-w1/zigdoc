pub fn init(ints: [int_count]Int) Self {

Set all entries of a packed array to the same value src
pub fn setAll(self: *Self, int: Int) void {
    var i: usize = 0;
    while (i < int_count) : (i += 1) {
        self.set(i, int);
    }
}

Calculates the number of bytes required to store a desired count of Ints src
pub fn bytesRequired(int_count: usize) usize {
    const total_bits = int_bits * int_count;
    const total_bytes = (total_bits + 7) / 8;
    return total_bytes;
}

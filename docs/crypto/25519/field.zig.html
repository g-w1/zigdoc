Return true if the field element is zero src
pub fn isZero(fe: Fe) callconv(.Inline) bool {
    var reduced = fe;
    reduced.reduce();
    const limbs = reduced.limbs;
    return (limbs[0] | limbs[1] | limbs[2] | limbs[3] | limbs[4]) == 0;
}

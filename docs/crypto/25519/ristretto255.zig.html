pub fn fromUniform(h: [64]u8) Ristretto255 {
    const p0 = elligator(Fe.fromBytes(h[0..32].*));
    const p1 = elligator(Fe.fromBytes(h[32..64].*));
    return Ristretto255{ .p = p0.add(p1) };
}

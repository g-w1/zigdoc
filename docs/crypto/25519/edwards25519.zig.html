pub fn toBytes(p: Edwards25519) [encoded_length]u8 {
    const zi = p.z.invert();
    var s = p.y.mul(zi).toBytes();
    s[31] ^= @as(u8, @boolToInt(p.x.mul(zi).isNegative())) << 7;
    return s;
}

pub fn clampedMul(p: Edwards25519, s: [32]u8) Error!Edwards25519 {
    var t: [32]u8 = s;
    scalar.clamp(&t);
    return mul(p, t);
}

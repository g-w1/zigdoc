pub fn ensureUnusedCapacity(self: *Self, size: usize) error{OutOfMemory}!void {
    if (self.writableLength() >= size) return;

    return try self.ensureCapacity(math.add(usize, self.count, size) catch return error.OutOfMemory);
}

pub fn eql(a: Address, b: Address) bool {
    const a_bytes = @ptrCast([*]const u8, &a.any)[0..a.getOsSockLen()];
    const b_bytes = @ptrCast([*]const u8, &b.any)[0..b.getOsSockLen()];
    return mem.eql(u8, a_bytes, b_bytes);
}

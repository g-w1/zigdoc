A lock that supports one writer or many readers. This API is for kernel threads, not evented I/O. This API requires being initialized at runtime, and initialization can fail. Once initialized, the core operations cannot fail. src

pub fn unlockShared(rwl: *DefaultRwLock) void {
    const state = @atomicRmw(usize, &rwl.state, .Sub, READER, .SeqCst);

    if ((state & READER_MASK == READER) and (state & IS_WRITING != 0))
        rwl.semaphore.post();
}

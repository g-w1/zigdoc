Returns the return type `sqrt` will return given an operand of type `T`. 
pub fn Sqrt(comptime T: type) type {
    return switch (@typeInfo(T)) {
        .Int => |int| std.meta.Int(.unsigned, (int.bits + 1) / 2),
        else => T,
    };
}

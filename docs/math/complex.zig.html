Returns the product of two complex numbers. src
pub fn mul(self: Self, other: Self) Self {
    return Self{
        .re = self.re * other.re - self.im * other.im,
        .im = self.im * other.re + self.re * other.im,
    };
}

pub fn conjugate(self: Self) Self {

Re-interpret a generic socket address into an IP address. 
pub fn from(address: Socket.Address) ip.Address {
    return switch (address) {
        .ipv4 => |ipv4_address| .{ .ipv4 = ipv4_address },
        .ipv6 => |ipv6_address| .{ .ipv6 = ipv6_address },
    };
}
